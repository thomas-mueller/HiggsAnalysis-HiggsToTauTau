216 void CombineHarvester::MergeBinErrors(double bbb_threshold,
217  double merge_threshold) {
218  // Reduce merge_threshold very slightly to avoid numerical issues
219  // E.g. two backgrounds each with bin error 1.0. merge_threshold of
220  // 0.5 should not result in merging - but can do depending on
221  // machine and compiler
222  merge_threshold -= 1E-9 * merge_threshold;
223  auto bins = this->bin_set();
224  for (auto const& bin : bins) {
225  // unsigned bbb_added = 0;
226  // unsigned bbb_removed = 0;
227  CombineHarvester tmp = std::move(this->cp().bin({bin}).histograms());
228  if (tmp.procs_.size() == 0) continue;
229 
230  std::vector<std::unique_ptr<TH1>> h_copies(tmp.procs_.size());
231  for (unsigned i = 0; i < h_copies.size(); ++i) {
232  h_copies[i] = tmp.procs_[i]->ClonedScaledShape();
233  }
234 
235  for (int i = 1; i <= h_copies[0]->GetNbinsX(); ++i) {
236  double tot_bbb_added = 0.0;
237  std::vector<std::pair<double, TH1 *>> result;
238  for (unsigned j = 0; j < h_copies.size(); ++j) {
239  double val = h_copies[j]->GetBinContent(i);
240  double err = h_copies[j]->GetBinError(i);
241  if (val == 0.0 && err == 0.0) continue;
242  if (val == 0 || (err/val) > bbb_threshold) {
243  // bbb_added += 1;
244  tot_bbb_added += (err * err);
245  result.push_back(std::make_pair(err*err, h_copies[j].get()));
246  }
247  }
248  if (tot_bbb_added == 0.0) continue;
249  std::sort(result.begin(), result.end());
250  double removed = 0.0;
251  for (unsigned r = 0; r < result.size(); ++r) {
252  if ((result[r].first + removed) < (merge_threshold * tot_bbb_added) &&
253  r < (result.size() - 1)) {
254  // bbb_removed += 1;
255  removed += result[r].first;
256  result[r].second->SetBinError(i, 0.0);
257  }
258  }
259  double expand = std::sqrt(1. / (1. - (removed / tot_bbb_added)));
260  for (unsigned r = 0; r < result.size(); ++r) {
261  result[r]
262  .second->SetBinError(i, result[r].second->GetBinError(i) * expand);
263  }
264  }
265  for (unsigned i = 0; i < h_copies.size(); ++i) {
266  tmp.procs_[i]->set_shape(std::move(h_copies[i]), false);
267  }
268  // std::cout << "BIN: " << bin << std::endl;
269  // std::cout << "Total bbb added: " << bbb_added << "\n";
270  // std::cout << "Total bbb removed: " << bbb_removed << "\n";
271  // std::cout << "Total bbb =======>: " << bbb_added-bbb_removed << "\n";
272  }
273 }
