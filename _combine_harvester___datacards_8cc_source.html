566  }

583  TH1::AddDirectory(add_dir);
584  break;
585  }
586  }
587  }
588  }
589  if (seen_lnU) {
590  line[1] = "lnU";
591  } else if (seen_lnN && !seen_shape) {
592  line[1] = "lnN";
593  } else if (!seen_lnN && seen_shape) {
594  line[1] = "shape";
595  } else if (seen_lnN && seen_shape) {
596  line[1] = "shape?";
597  } else {
598  throw std::runtime_error(FNERROR("Systematic type could not be deduced"));
599  }
600  txt_file << boost::format(
601  "%-"+sys_str_short+"s %-7s ")
602  % line[0] % line[1];
603  for (unsigned p = 0; p < procs_.size(); ++p) {
604  txt_file << boost::format("%-15s ") % line[p+2];
605  }
606  txt_file << "\n";
607  }
608  // write param line for any parameter which has a non-zero error
609  // and which doesn't appear in list of nuisances
610 
611 
612  std::set<std::string> ws_vars;
613  for (auto iter : wspaces_) {
614  RooArgSet vars = iter.second->allVars();
615  auto v = vars.createIterator();
616  do {
617  RooRealVar *y = dynamic_cast<RooRealVar*>(**v);
618  if (y) {
619  ws_vars.insert(y->GetName());
620  }
621  } while (v->Next());
622  }
623 
624  // How to check for params we need to write:
625  // Get a list of all pdf dependents for the processes we've just written
626  // Also need a list of all Systematic entries we've just written
627  // Then we write param if:
628  // - it has non-zero errors
629  // - it appears in the first list
630  // - it doesn't appear in the second list
631  for (auto par : params_) {

635  txt_file << boost::format("%-" +
636  boost::lexical_cast<std::string>(sys_str_len) +
637  "s param %g %g\n") %
638  p->name() % p->val() % ((p->err_u() - p->err_d()) / 2.0);
639  }
640  }
641  txt_file.close();
