178 double CombineHarvester::GetRateInternal(ProcSystMap const& lookup,
179  std::string const& single_sys) {
180  double rate = 0.0;
181  // TH1F tot_shape
182  for (unsigned i = 0; i < procs_.size(); ++i) {
183  double p_rate = procs_[i]->rate();
184  // If we are evaluating the effect of a single parameter
185  // check the list of associated nuisances and skip if
186  // this "single_sys" is not in the list
187  // However - we can't skip if the process has a pdf, as
188  // we haven't checked what the parameters are
189  if (single_sys != "" && !procs_[i]->pdf()) {
190  if (!ch::any_of(lookup[i], [&](Systematic const* sys) {
191  return sys->name() == single_sys;
192  })) continue;
193  }
194  for (auto sys_it : lookup[i]) {
195  double x = params_[sys_it->name()]->val();
196  if (sys_it->asymm()) {
197  if (x >= 0) {
198  p_rate *= std::pow(sys_it->value_u(), x * sys_it->scale());
199  } else {
200  p_rate *= std::pow(sys_it->value_d(), -1.0 * x * sys_it->scale());
201  }
202  } else {
203  p_rate *= std::pow(sys_it->value_u(), x * sys_it->scale());
204  }
205  }
206  rate += p_rate;
207  }

343 void CombineHarvester::ShapeDiff(double x,
344  TH1F * target,
345  TH1 const* nom,
346  TH1 const* low,
347  TH1 const* high,
348  bool linear) {
349  double fx = smoothStepFunc(x);
350  for (int i = 1; i <= target->GetNbinsX(); ++i) {
351  float h = high->GetBinContent(i);
352  float l = low->GetBinContent(i);
353  float n = nom->GetBinContent(i);
354  if (!linear) {
355  float t = target->GetBinContent(i);
356  target->SetBinContent(i, t > 0. ? std::log(t) : -999.);
357  h = (h > 0. && n > 0.) ? std::log(h/n) : 0.;
358  l = (l > 0. && n > 0.) ? std::log(l/n) : 0.;
359  target->SetBinContent(i, target->GetBinContent(i) +
360  0.5 * x * ((h - l) + (h + l) * fx));
361  target->SetBinContent(i, std::exp(target->GetBinContent(i)));
362  } else {
363  target->SetBinContent(i, target->GetBinContent(i) +
364  0.5 * x * ((h - l) + (h + l - 2. * n) * fx));
365  }
366  }
367 }

369 void CombineHarvester::ShapeDiff(double x,
370  TH1F * target,
371  RooDataHist const* nom,
372  RooDataHist const* low,
373  RooDataHist const* high) {
374  double fx = smoothStepFunc(x);
375  for (int i = 1; i <= target->GetNbinsX(); ++i) {
376  high->get(i-1);
377  low->get(i-1);
378  nom->get(i-1);
379  // The RooDataHists are not scaled to unity (unlike in the TH1 case above)
380  // so we have to normalise the bin contents on the fly
381  float h = high->weight() / high->sumEntries();
382  float l = low->weight() / low->sumEntries();
383  float n = nom->weight() / nom->sumEntries();
384  target->SetBinContent(i, target->GetBinContent(i) +
385  0.5 * x * ((h - l) + (h + l - 2. * n) * fx));
386  }

469 void CombineHarvester::VariableRebin(std::vector<double> bins) {
470  // We need to keep a record of the Process rates before we rebin. The
471  // reasoning comes from the following scenario: the user might choose a new
472  // binning which excludes some of the existing bins - thus changing the
473  // process normalisation. This is fine, but we also need to adjust the shape
474  // Systematic entries - both the rebinning and the adjustment of the value_u
475  // and value_d shifts.
476  std::vector<double> prev_proc_rates(procs_.size());
477 
478  // Also hold on the scaled Process hists *after* the rebinning is done - these
479  // are needed to update the associated Systematic entries
480  std::vector<std::unique_ptr<TH1>> scaled_procs(procs_.size());
481 
482  for (unsigned i = 0; i < procs_.size(); ++i) {
483  if (procs_[i]->shape()) {
484  // Get the scaled shape here
485  std::unique_ptr<TH1> copy(procs_[i]->ClonedScaledShape());
486  // shape norm should only be "no_norm_rate"
487  prev_proc_rates[i] = procs_[i]->no_norm_rate();
488  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
489  // The process shape & rate will be reset here
490  procs_[i]->set_shape(std::move(copy2), true);
491  scaled_procs[i] = procs_[i]->ClonedScaledShape();
492  }
493  }
494  for (unsigned i = 0; i < obs_.size(); ++i) {
495  if (obs_[i]->shape()) {
496  std::unique_ptr<TH1> copy(obs_[i]->ClonedScaledShape());
497  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
498  obs_[i]->set_shape(std::move(copy2), true);
499  }
500  }
501  for (unsigned i = 0; i < systs_.size(); ++i) {
502  TH1 const* proc_hist = nullptr;
503  double prev_rate = 0.;
504  for (unsigned j = 0; j < procs_.size(); ++j) {
505  if (MatchingProcess(*(procs_[j]), *(systs_[i].get()))) {
506  proc_hist = scaled_procs[j].get();
507  prev_rate = prev_proc_rates[j];
508  }
509  }
510  if (systs_[i]->shape_u() && systs_[i]->shape_d()) {
511  // These hists will be normalised to unity
512  std::unique_ptr<TH1> copy_u(systs_[i]->ClonedShapeU());
513  std::unique_ptr<TH1> copy_d(systs_[i]->ClonedShapeD());
514 
515  // If we found a matching Process we will scale this back up to their
516  // initial rates
517  if (proc_hist) {
518  copy_u->Scale(systs_[i]->value_u() * prev_rate);
519  copy_d->Scale(systs_[i]->value_d() * prev_rate);
520  }
521  std::unique_ptr<TH1> copy2_u(
522  copy_u->Rebin(bins.size() - 1, "", &(bins[0])));
523  std::unique_ptr<TH1> copy2_d(
524  copy_d->Rebin(bins.size() - 1, "", &(bins[0])));
525  // If we have proc_hist != nullptr, set_shapes will re-calculate value_u
526  // and value_d for us, before scaling the new hists back to unity
527  systs_[i]->set_shapes(std::move(copy2_u), std::move(copy2_d), proc_hist);
528  }
529  }
