43 double CombineHarvester::GetUncertainty() {
44  auto lookup = GenerateProcSystMap();

51  double rate_u = this->GetRateInternal(lookup, param_it.first);

137 double CombineHarvester::GetRateInternal(ProcSystMap const& lookup,
138  std::string const& single_sys) {
139  double rate = 0.0;
140  // TH1F tot_shape
141  for (unsigned i = 0; i < procs_.size(); ++i) {
142  double p_rate = procs_[i]->rate();
143  // If we are evaluating the effect of a single parameter
144  // check the list of associated nuisances and skip if
145  // this "single_sys" is not in the list
146  // However - we can't skip if the process has a pdf, as
147  // we haven't checked what the parameters are
148  if (single_sys != "" && !procs_[i]->pdf()) {
149  if (!ch::any_of(lookup[i], [&](Systematic const* sys) {
150  return sys->name() == single_sys;
151  })) continue;
152  }
153  for (auto sys_it : lookup[i]) {
154  double x = params_[sys_it->name()]->val();
155  if (sys_it->asymm()) {
156  if (x >= 0) {
157  p_rate *= std::pow(sys_it->value_u(), x * sys_it->scale());
158  } else {
159  p_rate *= std::pow(sys_it->value_d(), -1.0 * x * sys_it->scale());
160  }
161  } else {
162  p_rate *= std::pow(sys_it->value_u(), x * sys_it->scale());
163  }
164  }
165  rate += p_rate;
166  }
167  return rate;
168 }

335 void CombineHarvester::VariableRebin(std::vector<double> bins) {
336  // We need to keep a record of the Process rates before we rebin. The
337  // reasoning comes from the following scenario: the user might choose a new
338  // binning which excludes some of the existing bins - thus changing the
339  // process normalisation. This is fine, but we also need to adjust the shape
340  // Systematic entries - both the rebinning and the adjustment of the value_u
341  // and value_d shifts.
342  std::vector<double> prev_proc_rates(procs_.size());
343 
344  // Also hold on the scaled Process hists *after* the rebinning is done - these
345  // are needed to update the associated Systematic entries
346  std::vector<std::unique_ptr<TH1>> scaled_procs(procs_.size());
347 
348  for (unsigned i = 0; i < procs_.size(); ++i) {
349  if (procs_[i]->shape()) {
350  // Get the scaled shape here
351  std::unique_ptr<TH1> copy(procs_[i]->ClonedScaledShape());
352  // shape norm should only be "no_norm_rate"
353  prev_proc_rates[i] = procs_[i]->no_norm_rate();
354  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
355  // The process shape & rate will be reset here
356  procs_[i]->set_shape(std::move(copy2), true);
357  scaled_procs[i] = procs_[i]->ClonedScaledShape();
358  }
359  }
360  for (unsigned i = 0; i < obs_.size(); ++i) {
361  if (obs_[i]->shape()) {
362  std::unique_ptr<TH1> copy(obs_[i]->ClonedScaledShape());
363  std::unique_ptr<TH1> copy2(copy->Rebin(bins.size()-1, "", &(bins[0])));
364  obs_[i]->set_shape(std::move(copy2), true);
365  }
366  }
367  for (unsigned i = 0; i < systs_.size(); ++i) {
368  TH1 const* proc_hist = nullptr;
369  double prev_rate = 0.;
370  for (unsigned j = 0; j < procs_.size(); ++j) {
371  if (MatchingProcess(*(procs_[j]), *(systs_[i].get()))) {
372  proc_hist = scaled_procs[j].get();
373  prev_rate = prev_proc_rates[j];
374  }
375  }
376  if (systs_[i]->shape_u() && systs_[i]->shape_d()) {
377  // These hists will be normalised to unity
378  std::unique_ptr<TH1> copy_u(systs_[i]->ClonedShapeU());
379  std::unique_ptr<TH1> copy_d(systs_[i]->ClonedShapeD());
380 
381  // If we found a matching Process we will scale this back up to their
382  // initial rates
383  if (proc_hist) {
384  copy_u->Scale(systs_[i]->value_u() * prev_rate);
385  copy_d->Scale(systs_[i]->value_d() * prev_rate);
386  }
387  std::unique_ptr<TH1> copy2_u(
388  copy_u->Rebin(bins.size() - 1, "", &(bins[0])));
389  std::unique_ptr<TH1> copy2_d(
390  copy_d->Rebin(bins.size() - 1, "", &(bins[0])));
391  // If we have proc_hist != nullptr, set_shapes will re-calculate value_u
392  // and value_d for us, before scaling the new hists back to unity
393  systs_[i]->set_shapes(std::move(copy2_u), std::move(copy2_d), proc_hist);
394  }
395  }
396 }
