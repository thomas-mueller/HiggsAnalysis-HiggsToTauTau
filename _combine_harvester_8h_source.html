429 template <class Map>
430 void CombineHarvester::AddSyst(CombineHarvester& target,
431  std::string const& name, std::string const& type,
432  Map const& valmap) {
433  // Keep track of which Process entries get a Systematic assigned and which
434  // don't. If verbosity is on we'll print lists of these processes at the end.
435  std::vector<ch::Process *> not_added_procs;
436  std::vector<ch::Process *> added_procs;
437  // Also track which tuples in the map did not get used. Do this by getting the
438  // full map here and then removing elements as they are used to create a
439  // Systematic.
440  auto tuples = valmap.GetTupleSet();
441  if (verbosity_ >= 1) {
442  LOGLINE(log(), name + ":" + type);
443  }
444  for (unsigned i = 0; i < procs_.size(); ++i) {
445  if (!valmap.Contains(procs_[i].get())) {
446  not_added_procs.push_back(procs_[i].get());
447  continue;
448  }
449  tuples.erase(valmap.GetTuple(procs_[i].get()));
450  added_procs.push_back(procs_[i].get());
451  std::string subbed_name = name;
452  boost::replace_all(subbed_name, "$BIN", procs_[i]->bin());
453  boost::replace_all(subbed_name, "$PROCESS", procs_[i]->process());
454  boost::replace_all(subbed_name, "$MASS", procs_[i]->mass());
455  boost::replace_all(subbed_name, "$ERA", procs_[i]->era());
456  boost::replace_all(subbed_name, "$CHANNEL", procs_[i]->channel());
457  boost::replace_all(subbed_name, "$ANALYSIS", procs_[i]->analysis());
458  auto sys = std::make_shared<Systematic>();
459  ch::SetProperties(sys.get(), procs_[i].get());
460  sys->set_name(subbed_name);
461  sys->set_type(type);
462  if (type == "lnN" || type == "lnU") {
463  sys->set_asymm(valmap.IsAsymm());
464  sys->set_value_u(valmap.ValU(procs_[i].get()));
465  sys->set_value_d(valmap.ValD(procs_[i].get()));
466  } else if (type == "shape") {
467  sys->set_asymm(true);
468  sys->set_value_u(1.0);
469  sys->set_value_d(1.0);
470  sys->set_scale(valmap.ValU(procs_[i].get()));
471  }
472  CombineHarvester::CreateParameterIfEmpty(&target, sys->name());
473  if (sys->type() == "lnU") {
474  params_.at(sys->name())->set_err_d(0.);
475  params_.at(sys->name())->set_err_u(0.);
476  }
477  target.systs_.push_back(sys);
478  }
479  if (tuples.size() && verbosity_ >= 1) {
480  log() << ">> Map keys that were not used to create a Systematic:\n";
481  for (auto s : tuples) {
482  log() << ch::Tuple2String(s) << "\n";
483  }
484  }
485  if (verbosity_ >= 2) {
486  Process::PrintHeader(log());
487  log() << ">> Process entries that did not get a Systematic:\n";
488  for (auto p : not_added_procs) {
489  log() << *p << "\n";
490  }
491  log() << ">> Process entries that did get a Systematic:\n";
492  for (auto p : added_procs) {
493  log() << *p << "\n";
494  }
495  }
496 }
497 }
