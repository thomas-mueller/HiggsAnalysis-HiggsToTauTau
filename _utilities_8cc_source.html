139 std::vector<std::vector<unsigned>> GenerateCombinations(
140  std::vector<unsigned> vec) {
141  unsigned n = vec.size();
142  std::vector<unsigned> idx(n, 0);
143  std::vector<std::vector<unsigned>> result;
144  // if any one of the elements is zero there are no
145  // combinations to build
146  if (std::find(vec.begin(), vec.end(), 0) != vec.end()) {
147  return result;
148  }
149  result.push_back(idx);
150  bool exit_loop = false;
151  while (exit_loop == false) {
152  // Find the first index we can increment (if possible)
153  for (unsigned i = 0; i < n; ++i) {
154  if ((idx[i] + 1) == vec[i]) {
155  if (i != n - 1) {
156  idx[i] = 0;
157  } else {
158  // we're done
159  exit_loop = true;
160  break;
161  }
162  } else {
163  ++(idx[i]);
164  result.push_back(idx);
165  break;
166  }
167  }
168  }
169  return result;
170 }

189 std::vector<std::string> MassesFromRange(std::string const& input,
190  std::string const& fmt) {
191  std::set<double> mass_set;
192  std::vector<std::string> tokens;
193  boost::split(tokens, input, boost::is_any_of(","));
194  for (auto const& t : tokens) {
195  std::vector<std::string> sub_tokens;
196  boost::split(sub_tokens, t, boost::is_any_of("-:"));
197  if (sub_tokens.size() == 1) {
198  double mass_val = boost::lexical_cast<double>(sub_tokens[0]);
199  mass_set.insert(mass_val);
200  } else if (sub_tokens.size() == 3) {
201  double lo = boost::lexical_cast<double>(sub_tokens[0]);
202  double hi = boost::lexical_cast<double>(sub_tokens[1]);
203  double step = boost::lexical_cast<double>(sub_tokens[2]);
204  if (hi <= lo)
205  throw std::runtime_error(
206  "[MassesFromRange] High mass is smaller than low mass!");
207  double start = lo;
208  while (start < hi + 0.001) {
209  mass_set.insert(start);
210  start += step;
211  }
212  }
213  }
214  std::vector<std::string> result;
215  for (auto const& m : mass_set) {
216  result.push_back((boost::format(fmt) % m).str());
217  }
218  return result;
219 }
